\section{Internal categories}

This appendix is devoted to a quick recall of the definition of internal categories the main account are \cite{borceux1994handbook1,maclane2012sheaves,mac2013categories,jacobs1999categorical}.


\begin{definition}
	Let $\X$ be a category with pullbacks. An \emph{internal category} $\mathcal{C}$ in $\X$ is a $6$-uple $(O_{\mathcal{C}}, A_{\mathcal{C}}, s_{\mathcal{C}}, t_{\mathcal{C}},e_{\mathcal{C}},  c_{\mathcal{C}})$ given by
	\begin{itemize}
		\item two objects $O_{\mathcal{C}}$ and $A_{\mathcal{C}}$ called, respectively, the \emph{object of objects} and \emph{object of arrows} ;
		\item two \emph{source} and \emph{target} arrows $s_{\mathcal{C}}, t_{\mathcal{C}}\colon A_{\mathcal{C}}\rightrightarrows O_{\mathcal{C}}$;
		\item an \emph{identity arrow} $e_{\mathcal{C}}\colon O_{\mathcal{C}}\to A_{\mathcal{C}}$;
		\item a \emph{composition arrow} $P\to A_{\mathcal{C}}$, where $P$ is the vertex of a pullbacks square as the one below.
		\[\xymatrix{P \ar[r]^{p_1} \ar[d]_{p_0} &A_{\mathcal{C}} \ar[d]^{s_{\mathcal{C}}} \\ A_{\mathcal{C}} \ar[r]_{t_{\mathcal{C}}}  &O_{\mathcal{C}} }\]
	\end{itemize}

These data are required to satisfy the four axioms listed below.
\begin{enumerate}
	\item The following diagram commutes:
	\[\xymatrix{& O_{\mathcal{C}} \ar[d]_{e_{\mathcal{C}}}\ar[dr]^{\id{O_{\mathcal{C}}}} \ar[dl]_{\id{O_{\mathcal{C}}}}\\ O_{\mathcal{C}} & A_{\mathcal{C}}  \ar[l]^{t_{\mathcal{C}}}\ar[r]_{s_{\mathcal{C}}}& O_{\mathcal{C}}}\]
	\item The following two squares commute:
	\[\xymatrix{P \ar[r]^{c_{\mathcal{C}}} \ar[d]_{p_0}& A_{\mathcal{C}} \ar[d]^{s_{\mathcal{C}}} &P \ar[r]^{c_{\mathcal{C}}} \ar[d]_{p_1}& A_{\mathcal{C}} \ar[d]^{t_{\mathcal{C}}} \\ A_{\mathcal{C}} \ar[r]_{s_{\mathcal{C}}}& O_{\mathcal{C}} & A_{\mathcal{C}} \ar[r]_{t_{\mathcal{C}}}& O_{\mathcal{C}} }\]
		\item The following diagram commute:
	\[\xymatrix{A_{\mathcal{C}} \ar[dr]_{\id{A_{\mathcal{C}}}}\ar[r]^{\phi}& P \ar[d]_{c}&A_{\mathcal{C}} \ar[l]_{\phi'} \ar[dl]^{\id{A_{\mathcal{C}}}} \\ & A_{\mathcal{C}}}\]
	where $\phi, \phi'\colon A_{\mathcal{C}}\rightrightarrows P$ are the arrows defined by the two diagrams below, whose solid parts commute by the first axiom;
	\[\xymatrix{A_{\mathcal{C}} \ar@/_.3cm/[ddr]_{\id{A_{\mathcal{C}}}} \ar[r]^{t_{\mathcal{C}}}\ar@{.>}[dr]^{\phi} & O_{\mathcal{C}} \ar[dr]^{e_{\mathcal{C}}}  &&A_{\mathcal{C}} \ar@/^.3cm/[drr]^{\id{A_{\mathcal{C}}}} \ar[d]_{s_{\mathcal{C}}}\ar@{.>}[dr]^{\phi'} \\ & P \ar[r]^{p_1} \ar[d]_{p_0}& A_{\mathcal{C}} \ar[d]^{s_{\mathcal{C}}} &  O_{\mathcal{C}} \ar[dr]_{e_{\mathcal{C}}} & P \ar[r]^{p_1} \ar[d]_{p_0}& A_{\mathcal{C}} \ar[d]^{s_{\mathcal{C}}} \\ & A_{\mathcal{C}} \ar[r]_{t_{\mathcal{C}}}& O_{\mathcal{C}} & &A_{\mathcal{C}} \ar[r]_{t_{\mathcal{C}}}& O_{\mathcal{C}} }\]

\item Let us consider the following two pullback squares:
	\[\xymatrix{Q \ar[rr]^{q_1} \ar[d]_{q_0} & & A_{\mathcal{C}} \ar[d]^{s_{\mathcal{C}}}& Q' \ar[rr]^{q'_1} \ar[d]_{q'_0} &&  A_{\mathcal{C}} \ar[d]^{t_{\mathcal{C}}}\\ P   \ar[r]_{c} & A_{\mathcal{C}} \ar[r]_{t_{\mathcal{C}}} & O_{\mathcal{C}} & P \ar[r]_{c}& A_{\mathcal{C}} \ar[r]_{s_{\mathcal{C}}} & O_{\mathcal{C}}}\]
	
	
	
	
	then we ask for the commutativity of the diagram below.
	
\end{enumerate}
\end{definition}
